//TB T flip-flop
module top_module ();
    reg clk, reset , t;
    wire q ;
    
    tff uut ( .clk(clk), .reset(reset),.t(t),.q(q));
    
    
    initial begin
        clk = 1'b0;
        reset = 1'b0 ;
    end
    always #5 clk = ~clk;
    initial begin
        #5 reset = 1'b1 ;
        #20 reset = 1'b0;
    end
    always @(posedge clk) begin
        if(reset) t <= 1'b0;
        else t <= 1'b1;
    end
    

endmodule
